var acc = document.getElementsByClassName("accordion"); finds all the elements on the page that have a class name "accordion" and stores them in a variable. 
var i; 

for (i = 0; i < acc.length; i++) {
  acc[i].addEventListener("click", function() { Checks through all the accordions to see if they are being pressed.
  this.classList.toggle("active"); marks the accordion as active
  var panel = this.nextElementSibling;  Checks if the part after it is open or closed and reverses it
  if (panel.style.maxHeight) {
    panel.style.maxHeight = null;
  } else {
    panel.style.maxHeight = panel.scrollHeight + "px";
  } 
  });
}

function RemakeFields(event) {

    const nef_wrapper = document.createElement("div");

    const NameLabel = document.createElement("label");Creates all the parts that will be used in the button.
    const NameInput = document.createElement("input");

    const WeightLabel = document.createElement("Label");
    const WeightInput = document.createElement("input");

    const GradeLabel = document.createElement("label");
    const GradeInput = document.createElement("input");

    const btnAdd = document.createElement("button");
    const btnDel = document.createElement("button");

    nef_wrapper.classList.add("input__w");This goes around everything to be easily worked with.

    NameLabel.htmlFor = "Classname"; Here we give the labels a usage and a text.
    NameLabel.innerText = " Class Name: ";

    WeightLabel.htmlFor = "ClassWeight";
    WeightLabel.innerText = " Class Weight: ";

    GradeLabel.htmlFor = "FinalGrade";
    GradeLabel.innerText = " Final Grade: ";

    const currentFieldCounter = fieldCounter;Since numerous fields are being added we number them.
    NameInput.type = "text";Now we give each input its own data
    NameInput.name = "ClassName[]";
    NameInput.id = "ClassName" + currentFieldCounter;
    NameInput.setAttribute("required", "");
    NameInput.classList.add("input-field");

    WeightInput.type = "number";
    WeightInput.name = "ClassWeight[]";
    WeightInput.id = "ClassWeight" + currentFieldCounter;
    WeightInput.step = "0.1";
    WeightInput.max = "5";
    WeightInput.setAttribute("required", "");
    WeightInput.classList.add("input-field");

    GradeInput.type = "text";
    GradeInput.name = "FinalGrade[]";
    GradeInput.id = "FinalGrade" + currentFieldCounter;
    GradeInput.step = "1";
    GradeInput.max = "110";
    GradeInput.setAttribute("required", "");
    GradeInput.classList.add("input-field");

    btnDel.type = "button"; Adds a delete button.
    btnDel.classList.add("btn-del-input");
    btnDel.innerText = "-";

    nef_wrapper.appendChild(NameLabel);Adds all the buttons and labels to the wrapper
    nef_wrapper.appendChild(NameInput);
    nef_wrapper.appendChild(WeightLabel);
    nef_wrapper.appendChild(WeightInput);
    nef_wrapper.appendChild(GradeLabel);
    nef_wrapper.appendChild(GradeInput);
    nef_wrapper.appendChild(btnDel);

    myForm.appendChild(nef_wrapper);Adds the wrapper to the main page
    btnDel.addEventListener("click", removeEmailField); When the delete button is clicked it will remove the wrapper and everything inside it
    fieldCounter++;
    function removeEmailField(el) {
    const field = el.target.parentElement;
    field.remove();
    saveFormDataToCookie(); Save the data to be stored
}
